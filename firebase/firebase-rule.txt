rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read: if request.auth != null && (request.auth.uid == userId || request.auth.token.role == 'admin');
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /diagnosis/{docId} {
      allow create: if request.auth != null && request.resource.data.createdBy == request.auth.uid;
      allow read: if request.auth != null && (request.auth.token.role == 'admin' || resource.data.createdBy == request.auth.uid);
      allow update, delete: if request.auth != null && request.auth.token.role == 'admin';
    }
  }
}
// patients: user-created patient profiles
match /patients/{pid} {
  allow create: if request.auth != null;
  allow read: if request.auth != null && (request.auth.token.role == 'admin' || resource.data.createdBy == request.auth.uid);
  allow update, delete: if request.auth != null && request.auth.token.role == 'admin';
}

// model_metrics: only admin read/write
match /model_metrics/{mid} {
  allow create: if request.auth != null; // clients can submit metrics (or restrict to server)
  allow read, update, delete: if request.auth != null && request.auth.token.role == 'admin';
}
